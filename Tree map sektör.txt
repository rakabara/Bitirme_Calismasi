Tree map sektör

library(ggplot2)
install.packages("treemapify")
library(treemapify) # Ağaç haritası eklentisi
library(dplyr)
library(tidyr)

# Windows sistemlerde Türkçe karakter sorununu önleyen komut
Sys.setlocale("LC_ALL", "Turkish")

# 1. VERİYİ İÇERİ AKTARMA VE TEMİZLEME
# CSV dosyanızı okutuyoruz (Noktalı virgül ayrımına göre)
df <- read.csv2("sektorler.csv")

# Sadece sektörleri bırakıp, 13 yıllık toplam ölümleri hesaplıyoruz
df_sektor <- df %>%
  select(-Yil, -Bilinmeyen, -Toplam) %>%
  summarise_all(sum) %>%
  pivot_longer(cols = everything(), names_to = "Sektor", values_to = "Olum") %>%
  arrange(desc(Olum)) # Büyükten küçüğe sıralama

# Görselde estetik durması için etiket isimlerindeki alt tireleri temizliyoruz
df_sektor$Sektor <- gsub("_", " ", df_sektor$Sektor)
df_sektor$Sektor <- gsub("İsler", "İşler", df_sektor$Sektor)

# 2. VERİ GAZETECİLİĞİ ODAKLI AĞAÇ HARİTASI (TREEMAP) TASARIMI
ggplot(df_sektor, aes(area = Olum, fill = Olum, label = paste0(Sektor, "\n", Olum))) +
  
  # Blokların çizimi ve aralarındaki estetik beyaz sınırlar
  geom_treemap(color = "white", size = 2) +
  
  # Blok içindeki metinlerin ayarlanması (Otomatik ortalanır)
  geom_treemap_text(color = "white", place = "centre", grow = FALSE, fontface = "bold", size = 13) +
  
  # SIRALI RENK (Sequential Color): Düşük ölümden (açık renk) yüksek ölüme (koyu bordo) renk geçişi
  scale_fill_gradient(low = "#ffbaba", high = "#8b0000", guide = "none") +
  
  # Editoryal Bağlam ve Başlıklar
  labs(
    title = "İş Cinayetlerinin Sektörel Hiyerarşisi (2013-2025)",
    subtitle = "Dikdörtgenlerin büyüklüğü ve renginin koyuluğu, sektörlerdeki ölüm hacmini yansıtmaktadır.\nİnşaat, Tarım ve Taşımacılık 13 yıllık tabloda en büyük risk alanları olarak öne çıkıyor.",
    
  ) +
  
  # Temiz ve Modern Tema
  theme_minimal(base_size = 14) +
  theme(
    plot.title = element_text(face = "bold", size = 20, color = "#2c3e50", margin = margin(b = 8)),
    plot.subtitle = element_text(size = 14, color = "#34495e", margin = margin(b = 20)),
    plot.caption = element_text(size = 10, color = "grey50", hjust = 0, margin = margin(t = 15)),
    plot.margin = margin(20, 20, 20, 20)
  )
